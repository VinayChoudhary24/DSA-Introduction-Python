#Mutability: when we create a variable, it gets a location in memory and when we try to update it on
the same location in memory it is called mutability.

#Immutability: when we create a variable, it gets a location in memory and when we try to update, 
it gets a new location in memory again it is called immutability.

#Mutable/Immutable Example: when we write something on a paper using pencil we can erase it and write on the 
same line and place i.e mutability. if we use a pen we cannot erase it and we need to write on a different 
place i.e immutability.

##In Python Lists/Arrays are Mutable.

###Question
Find the Character in the string i.e how many times a character is repeated

Answer:
def check_char_freq(string, char):
    idx = 0
    count = 0
    while idx < len(string):
        if string[idx] == char:
            count += 1
        idx += 1
    return count

check_char_freq("happy birthday", "h")

Answer using for Loop:
def check_char_freq(string, char):
    count = 0
    for idx in range(0, len(string)):
        if string[idx] == char:
            count += 1
        idx += 1
    return count

Answer: Another Version:

###Question
Check if the given string is Palindrome or not

Answer:
def check_palindrome(string):
    n = len(string) #length of string
    i = 0 # low pointer starts from left most
    j = n - 1 # high pointer starts from right most
    while i <= j: # we need to make sure we dont loop if low and high cross each other 
        if string[i] != string[j]:
            return False
        i += 1 # move low pinter to the right
        j -= 1 # move high pointer to the left
    return True

check_palindrome("naman")

###Question
Problem statement
For a given input string(str), find and return the total number of words present in it.

It is assumed that two words will have only a single space in between. Also, there wouldn't be any 
leading and trailing spaces in the given input string.

Answer Using Built-in Function:
def countWords(string):
    if len(string) == 0:
        return 0
    return len(string.split(" "))

print(countWords("Hello World India"))


Answer Using count words:
def countWords(string):
    # If the string is empty, return 0
    if len(string) == 0:
        return 0
    
    # Count the number of spaces and add 1
    # (number of words = number of spaces + 1)
    count = 1
    for char in string:
        if char == ' ':
            count += 1
    
    return count

print(countWords("Hello World India"))



###Question
Print each word in sentence in reverse order 

Answer:
def reverseEachWord(string):
    if len(string) == 0:
        return 0
    rev = string.split(" ")
    result = ""
    for word in rev:
        result += word[::-1]
        result += " "
    return result[:-1]

print(reverseEachWord("This is"))



###Question
Show a Example of String Compression

Answer:
def string_compression(s):
    if s == "":
        return s
    result = ""
    count = 1
    for i in range(1, len(s)):
        if s[i] == s[i-1]:
            count += 1
        else:
            result += (s[i-1] + (str(count) if count > 1 else ""))
            count = 1

    result += (s[-1] + (str(count) if count > 1 else ""))
    return result

print(string_compression("aaaabbbceed"))

    # write your code here !!!
    rev2 = str2[::-1]
    rev1 = str1[::-1]
    if rev2 == str1:
        return True
    elif rev1 == str2:
        return True
    else:

###Question
Given a string and a character X, write a function to remove all occurrences of the character X from the string.

Answer:
def remove_all_occurrences_of_char(input_string, char_to_remove):
    result = ""
    for char in input_string:
        if char != char_to_remove:
            result += char
    return result

result = remove_all_occurrences_of_char("adasadsdsd", "s")
print(result)



###Question
You are given a string 'STR' consisting of lower and upper case characters. You need to remove the consecutive 
duplicates characters, and return the new string.

Answer:
def removeConsecutiveDuplicates(string) :
    #write your code logic here !!!
    if len(string) == 0:
        return ""
    result = string[0]
    for i in range(1, len(string)):
        if string[i] != string[i-1]:
            result += string[i]
    return result

new_string = removeConsecutiveDuplicates("aabbcssddaddd")
print(new_string)

###Question
Given a string ‘str’ and a pattern ‘pat’, you have to find all occurrences of the pattern in the string. 
You have to print the starting positions of all occurrences of the pattern in the string.

Answer:
def findSubstringIndices(string, pattern):
    indices = []
    n = len(string)
    m = len(pattern)
    
    for i in range(n - m + 1):
        # print(i, i+m)
        if string[i:i + m] == pattern:
            indices.append(i)
    
    return indices

result = findSubstringIndices("hidhdvnshidnmasndndhi", "hi")
print(result)
