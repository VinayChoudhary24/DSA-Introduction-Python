###Question
Get the unique Elements from the tuple

Answer:
def get_unique_ele(tup):
    result_tup = []
    for ele in tup:
        if ele not in result_tup:
            result_tup.append(ele)
    return tuple(result_tup)

unique_tuple = (2, 1, 7, 2, 9, 7, 4)
unique_tuple_res = get_unique_ele(unique_tuple)
print(unique_tuple_res)

###Question
Given a tuple of integers, your task is to implement a function that returns the maximum and 
minimum elements in the tuple. 

Answer:
def find_max_min(data):
    max_res = data[0]
    min_res = data[0]
    for ele in data:
        max_res = max(max_res, ele)
        min_res = min(min_res, ele)
    return max_res, min_res

max_min_tuple = (2, 1, 7, 2, 9, 7, 4)
max_min_tuple_res = find_max_min(max_min_tuple)
print(max_min_tuple_res)

(9, 1)

###Question
In a given string print all the vowels present in the string

Answer:
def check_vowels(string):
    vowel_set = {"a", "e", "i", "o", "u",}
    result = set()
    for i in range(0, len(string)):
        if string[i] in vowel_set:
            result.add(string[i])
    return result

string = "vinay"
vow_res = check_vowels(string)
print(vow_res)

###Question
Check if subarray sum is equal to zero in a list. if present return true or fasle

Answer:
# First we calculate the prefix_sum and than we check the if the sum is already present in set
# if present we know that the subarray sum is zero 
def check_subarray_sum_zero(li):
    if len(li) == 0:
        return False
    if li[0] == 0:
        return True
    s = {li[0]}
    prefix_sum = li[0]
    for i in range(1, len(li)):
        prefix_sum += li[i]
        if prefix_sum in s or prefix_sum == 0:
            return True
        s.add(prefix_sum)
    return False

subarray_li = [1, 2, 2, -4, 3, 4]
subarray_res = check_subarray_sum_zero(subarray_li)
print(subarray_res)

True

###Question
Count Vowels, Consonants and Spaces

Given a string, write a program to count the number of vowels, consonants, and spaces in that string.

Answer:
def countVowelsConsonantsSpaces(s: str, n: int):
    vowels = 0
    consonants = 0
    spaces = 0
    
    # Define vowel set for quick lookup
    vowel_set = set('aeiouAEIOU')
    
    for char in s:
        if char == ' ':
            spaces += 1
        elif char.isalpha():
            if char in vowel_set:
                vowels += 1
            else:
                consonants += 1
    
    return [vowels, consonants, spaces]

count_string = "hello world"
count_string_len = len(count_string)
count_string_result = countVowelsConsonantsSpaces(count_string, count_string_len)
print(count_string_result)

[3, 7, 1]
