## Elif Condition
marks = int(input())
if marks >= 90 and marks <= 100:
    print("This is If Condition")
elif marks >= 80 and marks <= 89:
    print("This is First Elif Condition")
elif marks >= 50 and marks <= 79:
    print("This is Second Elif Condition")
else:
    print("This is Else Condition")

## Nested if Condition
marks = int(input())
if marks >= 90 and marks <= 100:
    print("This is If Condition")
    if marks == 100:
        print("Full Marks")
elif marks >= 80 and marks <= 89:
    print("This is First Elif Condition")
elif marks >= 50 and marks <= 79:
    print("This is Second Elif Condition")
else:
    print("This is Else Condition")
