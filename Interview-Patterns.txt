 12321

"""

    #Second Loop is for printing Spaces
    while spaces <= n - i:
        print(" ", end="")

###Question
"""
Diamond of stars
Note: N is always odd.

Pattern for N = 5
  ***
 *****
  ***
   *
"""

Answer:
n = int(input())  # e.g. 5 -> middle row has 5 stars

# Upper half (including middle)
i = 1
while i <= (n + 1) // 2:
    spaces = 1
    while spaces <= ((n + 1) // 2) - i:
        print(" ", end="")
        spaces += 1

    stars = 1
    while stars <= (2 * i - 1):
        print("*", end="")
        stars += 1

    print()
    i += 1

# Lower half
i = (n + 1) // 2 - 1
while i >= 1:
    spaces = 1
    while spaces <= ((n + 1) // 2) - i:
        print(" ", end="")
        spaces += 1

    stars = 1
    while stars <= (2 * i - 1):
        print("*", end="")
        stars += 1

    print()
    i -= 1

5432*

*4321
"""

i = 1
while i <= n:
    j = 1
    while j <= n:
        # When column matches current row â†’ print '*'
        if j == n - i + 1:
            print('*', end="")
        else:

        j += 1
    print()

"""

Answer:
def ninjaPuzzle(n):
    i = 1
    while i <= n:
        # Print leading spaces
        spaces = 1
        while spaces < i:
            print(" ", end="")
            spaces += 1

        # Print stars
        stars = n - i + 1
        while stars > 0:
            print("*", end="")
            stars -= 1

        print()  # Move to next line
        i += 1
